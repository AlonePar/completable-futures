41    /**
42     * Returns a new {@link CompletableFuture} which completes to a list of all values of its input
43     * stages, if all succeed.  The list of results is in the same order as the input stages.
44     *
45     * <p>If any of the given stages complete exceptionally, then the returned future also does so,
46     * with a CompletionException holding this exception as its cause.
47     *
48     * <p>If no stages are provided, returns a future holding an empty list.
49     *
50     * @param stages The stages to combine.
51     * @return A future that completes to a list of the results of the supplied stages.
52     * @throws NullPointerException if the stages list or any of its elements are {@code null}.
53     */
54    public static <T> CompletableFuture<List<T>> allAsList(
55        List<? extends CompletionStage<? extends T>> stages) {
56      // We use traditional for-loops instead of streams here for performance reasons,
57      // see AllAsListBenchmark
58  
59      @SuppressWarnings("unchecked") // generic array creation
60      final CompletableFuture<? extends T>[] all = new CompletableFuture[stages.size()];
61      for (int i = 0; i < stages.size(); i++) {
62        all[i] = stages.get(i).toCompletableFuture();
63      }
64      return CompletableFuture.allOf(all)
65          .thenApply(ignored -> {
66            final List<T> result = new ArrayList<>(all.length);
67            for (int i = 0; i < all.length; i++) {
68              result.add(all[i].join());
69            }
70            return result;
71          });
72    }

74    /**
75     * Returns a new {@code CompletableFuture} that is already exceptionally completed with
76     * the given exception.
77     *
78     * @param throwable the exception
79     * @return a future that exceptionally completed with the supplied exception
80     * @throws NullPointerException if the supplied throwable is {@code null}
81     */
82    public static <T> CompletableFuture<T> exceptionallyCompletedFuture(Throwable throwable) {
83      final CompletableFuture<T> future = new CompletableFuture<>();
84      future.completeExceptionally(throwable);
85      return future;
86    }

88    /**
89     * Collect a stream of {@link CompletionStage}s into a single future holding a list of the
90     * joined entities.
91     * Usage:
92     * <pre>
93     *   collection.stream()
94     *      .map(this::someAsyncFunc)
95     *      .collect(joinList())
96     *      .thenApply(this::consumeList)
97     * </pre>
98     * The generated CompletableFuture will complete to a list of all entities, in the order they were
99     * encountered in the original stream. Similar to
100    * {@link CompletableFuture#allOf(CompletableFuture[])}, if any of the input futures complete
101    * exceptionally, then the returned CompletableFuture also does so, with a CompletionException
102    * holding this exception as its cause.
103    *
104    * @throws NullPointerException if any future in the stream is {@code null}.
105    */
106   public static <T, S extends CompletionStage<? extends T>>
107   Collector<S, ?, CompletableFuture<List<T>>> joinList() {
108     return collectingAndThen(toList(), CompletableFutures::allAsList);
109   }

111   /**
112    * Check that a stage is completed.
113    * @param stage a {@link CompletionStage}.
114    * @throws IllegalStateException if the stage is not completed.
115    */
116   public static <T> void checkCompleted(CompletionStage<T> stage) {
117     if (!stage.toCompletableFuture().isDone()) {
118       throw new IllegalStateException("future was not completed");
119     }
120   }

122   /**
123    * Get the value of a completed stage.
124    *
125    * @param stage a completed {@link CompletionStage}.
126    * @return the value of the stage if it has one.
127    * @throws IllegalStateException if the stage is not completed.
128    */
129   public static <T> T getCompleted(CompletionStage<T> stage) {
130     CompletableFuture<T> future = stage.toCompletableFuture();
131     checkCompleted(future);
132     return future.join();
133   }

135   /**
136    * Returns a new stage that, when this stage completes
137    * either normally or exceptionally, is executed with this stage's
138    * result and exception as arguments to the supplied function.
139    *
140    * <p>When this stage is complete, the given function is invoked
141    * with the result (or {@code null} if none) and the exception (or
142    * {@code null} if none) of this stage as arguments, and the
143    * function's result is used to complete the returned stage.
144    *
145    * <p>This differs from
146    * {@link java.util.concurrent.CompletionStage#handle(java.util.function.BiFunction)}
147    * in that the function should return a {@link java.util.concurrent.CompletionStage} rather than
148    * the value directly.
149    *
150    * @param stage the {@link CompletionStage} to compose
151    * @param fn the function to use to compute the value of the
152    * returned {@link CompletionStage}
153    * @param <U> the function's return type
154    * @return the new {@link CompletionStage}
155    */
156   public static <T, U> CompletionStage<U> handleCompose(
157       CompletionStage<T> stage,
158       BiFunction<? super T, Throwable, ? extends CompletionStage<U>> fn) {
159     return dereference(stage.handle(fn));
160   }
161 
162   /**
163    * Returns a new stage that, when this stage completes
164    * exceptionally, is executed with this stage's exception as the
165    * argument to the supplied function.  Otherwise, if this stage
166    * completes normally, then the returned stage also completes
167    * normally with the same value.
168    *
169    * <p>This differs from
170    * {@link java.util.concurrent.CompletionStage#exceptionally(java.util.function.Function)}
171    * in that the function should return a {@link java.util.concurrent.CompletionStage} rather than
172    * the value directly.
173    *
174    * @param stage the {@link CompletionStage} to compose
175    * @param fn the function to use to compute the value of the
176    * returned {@link CompletionStage} if this stage completed
177    * exceptionally
178    * @return the new {@link CompletionStage}
179    */
180   public static <T> CompletionStage<T> exceptionallyCompose(
181       CompletionStage<T> stage,
182       Function<Throwable, ? extends CompletionStage<T>> fn) {
183     return dereference(wrap(stage).exceptionally(fn));
184   }
185 
186   /**
187    * This takes a stage of a stage of a value and returns a plain stage of a value.
188    *
189    * @param stage a {@link CompletionStage} of a {@link CompletionStage} of a value
190    * @return the {@link CompletionStage} of the value
191    */
192   public static <T> CompletionStage<T> dereference(
193       CompletionStage<? extends CompletionStage<T>> stage) {
194     return stage.thenCompose(Function.identity());
195   }

197   private static <T> CompletionStage<CompletionStage<T>> wrap(CompletionStage<T> future) {
198     //noinspection unchecked
199     return future.thenApply(CompletableFuture::completedFuture);
200   }

202   public static <R, A, B> CompletionStage<R> combine(
203       CompletionStage<A> a, CompletionStage<B> b,
204       BiFunction<A, B, R> function) {
205     return a.thenCombine(b, function);
206   }

208   public static <R, A, B, C> CompletionStage<R> combine(
209       CompletionStage<A> a, CompletionStage<B> b, CompletionStage<C> c,
210       Function3<A, B, C, R> function) {
211     final CompletableFuture<A> af = a.toCompletableFuture();
212     final CompletableFuture<B> bf = b.toCompletableFuture();
213     final CompletableFuture<C> cf = c.toCompletableFuture();
214 
215     return CompletableFuture.allOf(af, bf, cf)
216         .thenApply(ignored -> function.apply(af.join(), bf.join(), cf.join()));
217   }

219   public static <R, A, B, C, D> CompletionStage<R> combine(
220       CompletionStage<A> a, CompletionStage<B> b, CompletionStage<C> c, CompletionStage<D> d,
221       Function4<A, B, C, D, R> function) {
222     final CompletableFuture<A> af = a.toCompletableFuture();
223     final CompletableFuture<B> bf = b.toCompletableFuture();
224     final CompletableFuture<C> cf = c.toCompletableFuture();
225     final CompletableFuture<D> df = d.toCompletableFuture();
226 
227     return CompletableFuture.allOf(af, bf, cf, df)
228         .thenApply(ignored -> function.apply(af.join(), bf.join(), cf.join(), df.join()));

231   public static <R, A, B, C, D, E> CompletionStage<R> combine(
232       CompletionStage<A> a, CompletionStage<B> b, CompletionStage<C> c,
233       CompletionStage<D> d, CompletionStage<E> e,
234       Function5<A, B, C, D, E, R> function) {
235     final CompletableFuture<A> af = a.toCompletableFuture();
236     final CompletableFuture<B> bf = b.toCompletableFuture();
237     final CompletableFuture<C> cf = c.toCompletableFuture();
238     final CompletableFuture<D> df = d.toCompletableFuture();
239     final CompletableFuture<E> ef = e.toCompletableFuture();
240 
241     return CompletableFuture.allOf(af, bf, cf, df, ef)
242         .thenApply(ignored ->
243                        function.apply(af.join(), bf.join(), cf.join(), df.join(), ef.join()));
244   }

246   /**
247    * Periodically poll an external resource until it returns a non-empty result.
248    *
249    * <p>The polling task should return Optional.empty() until it becomes available, and then
250    * Optional.of(result). If the polling task throws an exception or returns null, that will cause
251    * the result future to complete exceptionally.
252    *
253    * <p>Canceling the returned future will cancel the scheduled polling task as well.
254    *
255    * <p>Note that on a ScheduledThreadPoolExecutor the polling task might remain allocated for up
256    * to 'frequency' time after completing or being cancelled. If you have lots of polling operations
257    * or a long polling frequency, consider setting removeOnCancelPolicy to true.
258    * See {@link java.util.concurrent.ScheduledThreadPoolExecutor#setRemoveOnCancelPolicy(boolean)}.
259    *
260    * @param pollingTask the polling task.
261    * @param frequency the frequency to run the polling task at.
262    * @param executorService the executor service to schedule the polling task on.
263    * @return a future completing to the result of the polling task once that becomes available.
264    */
265   public static <T> CompletableFuture<T> poll(
266       final Supplier<Optional<T>> pollingTask,
267       final Duration frequency,
268       final ScheduledExecutorService executorService) {
269     final CompletableFuture<T> result = new CompletableFuture<>();
270     final ScheduledFuture<?> scheduled = executorService.scheduleAtFixedRate(
271         () -> pollTask(pollingTask, result), 0, frequency.toMillis(), TimeUnit.MILLISECONDS);
272     result.whenComplete((r, ex) -> scheduled.cancel(true));
273     return result;
274   }
275 
276   private static <T> void pollTask(
277       final Supplier<Optional<T>> pollingTask,
278       final CompletableFuture<T> resultFuture) {
279     try {
280       pollingTask.get().ifPresent(resultFuture::complete);
281     } catch (Exception ex) {
282       resultFuture.completeExceptionally(ex);
283     }
284   }
